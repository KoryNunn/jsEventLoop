eventQueue = []

fs.readFile = function(fileName, options, callback){
    error or file = readTheFile(fileName, options)

    eventQueue.push(() => callback(error, file))
}

runSource(mainProgram)

function run(nextStack){
    nextStack();

    if(eventQueue.length){
        run(eventQueue.shift())
    }
}
